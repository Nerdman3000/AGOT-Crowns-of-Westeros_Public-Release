# Crowns of Westeros Crown Scripted Effects
create_artifact_aegonvi_crown_effect = {
	# Get the character the artifact is being made for.
	$OWNER$ = { save_scope_as = owner }

	set_artifact_rarity_illustrious = yes

	# Create the artifact
	create_artifact = {	
		name = aegonvi_crown_name
		description = aegonvi_crown_description
		type = helmet
		visuals = aegonvi_crown_visuals
		wealth = scope:wealth
		quality = scope:quality
		template = aegonvi_crown_template
		modifier = aegonvi_crown_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}

	scope:newly_created_artifact = {
		set_variable = { name = crown_aegonvi_artifact value = yes }
		set_variable = { name = historical_unique_artifact value = yes }
		set_variable = {
			name = artifact_succession_title
			value = title:b_the_red_keep
		}
		add_artifact_title_history = {
			target = title:e_the_iron_throne
			date = 7998.1.2
		}
	}
	
	hidden_effect = {
		dynasty:dynn_Targaryen = {
			ordered_dynasty_member = {
				order_by = highest_held_title_tier
				house = { add_house_artifact_claim = scope:newly_created_artifact }
			}
		}
	}
}
